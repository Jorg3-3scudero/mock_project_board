Muestra de Código

@app.get("/api/estudiantes", response_model=list[EstudianteResponse])
async def get_estudiantes(
    programa: str | None = None,
    semestre: int | None = None,
    db: Session = Depends(get_db)
):
    """
    Retorna lista de estudiantes con filtros opcionales.
    Requiere autenticación JWT.
    """
    query = db.query(Estudiante)
    if programa:
        query = query.filter(Estudiante.programa == programa)
    if semestre:
        query = query.filter(Estudiante.semestre == semestre)
    return query.all()

Endpoint de ejemplo — FastAPI con filtros opcionales por query params, inyección de dependencias y tipado estricto con Pydantic.
